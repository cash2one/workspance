UICityList 模块是一个可定制数据源的城市列表。

var UICityList = api.require('UICityList');
UICityList.open({
    rect: {
        x: 20,
        y: 20,
        w: 280,
        h: 400
    },
    resource: "widget://res/cityList.json",
  // currentCity: "北京",
   locationWay: "wifi",
   hotTitle: "很热门城",
   placeholder: "名字，字母",
    styles: {
        searchBar: {
            bgColor: 'rgba(0,0,200,0.8)',
            cancelColor: 'rgba(200,0,0,0.7)'
        },
        location: {
            color: 'rgba(0,0,200,0.8)',
            size: 14
        },
        sectionTitle: {
            bgColor: 'rgba(0,0,55,0.4)',
            color: 'rgba(100,0,0,1)',
            size: 14
           // height: 40
        },
        item: {
            bgColor: 'rgba(0,200,0,0.8)',
            activeBgColor: 'rgba(0,0,200,0.9)',
            color: 'rgba(230,0,0,0.9)',
            size: 14
          //  height: 40
        },
        indicator: {
            bgColor: 'rgb(0,0,100)',
            color: 'rgb(200,0,0)'
        }
    }
}, function(ret, err){
   if(ret.eventType == 'selected'){
        alert(JSON.stringify(ret.cityInfo));
    }
});
