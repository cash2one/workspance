open 打开系统相册资源选择器

var mediaScanner = api.require('mediaScanner');

mediaScanner.open({
    bgColor: "#FFFFFF", // 背景色, 默认#FFFFFF, 支持rgb，rgba，#，imgPath.
    row: 4, // 图片显示的列数.默认 4.
    mark: {
        icon: "#696969", // 标记的图标,默认 #696969, 支持支持rgb，rgba，#，imgPath.
        position: "left_down", // 标记的位置, 默认left_down, 可选:left_up(左上角), left_down(左下角), right_up(右上角),right_down(右下角).
        size: undefined // 标记的大小,默认为默认图片宽度的三分之一.
    }, // 选中时显示的标记图标的相关配置.
    navigation: {
        bg: "rgba(0.5,0.5,0.5,0.8)",
        state: {
            title: "已选择*项",
            titleColor: "#0000FF",
            titleSize: 18
        },
        cancel:{
            title: "取消", // 标题,默认"取消".
            titleColor: "#0000FF", // 标题颜色.默认蓝色.支持rgb，rgba，#.
            titleSize: 18, // 字体大小.默认18.
            bg: "rgba(255, 255, 255, 0.6)" // 背景.默认: 白色 + 透明度 0.6.支持rgb，rgba，#，imgPath
        }, // 顶部导航设置
        finish: {
            title: "完成", // 标题,默认"完成".
            titleColor: "#0000FF", // 标题颜色.默认蓝色.支持rgb，rgba，#.
            titleSize: 18, // 字体大小.默认18.
            bg: "rgba(255, 255, 255, 0.6)" // 背景.默认: 白色 + 透明度 0.6.支持rgb，rgba，#，imgPath
        } // 底部导航设置
    }
},function(ret, err) {
    var list = ret.list;

    var msg = "";

    for(var idx in list){
        var item = list[idx];

        msg += "资源路径: " + item.url + "\n" +
            "缩略图路径: " + item.thumbUrl  +"\n" +
            "资源类型: " + item.mimeType  + "\n" +
            "资源大小: " + item.size + "\n\n\n"
    }

    alert(msg);
});

scann 获取系统相册所有媒体资源

/* 获取系统相册所有媒体资源 */

var mediaScanner = api.require('mediaScanner');

mediaScanner.scann(function(ret, err) {
    var list = ret.list;

    var msg = "";

    for(var idx in list){
        var item = list[idx];

        msg += "资源路径: " + item.url + "\n" +
            "缩略图路径: " + item.thumbUrl  +"\n" +
            "资源类型: " + item.mimeType  + "\n" +
            "资源大小: " + item.size + "\n\n\n"
    }

    alert(msg);
});
