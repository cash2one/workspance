arcColorPicker 可自定义样式、内容颜色的环形取色器

/* 打开取色器. */
var arcColorPicker = api.require('arcColorPicker');

arcColorPicker.open({
    gradients: ['#000000','#ff0000']
},function(ret, err){
    api.alert({msg: "open: " + JSON.stringify(ret) + JSON.stringify(err)});
});

/* 3m秒后,获取当期取色器的值.*/
setTimeout(function(ret, err){
    arcColorPicker.getColor(function(ret, err){
        api.alert({msg: "getColor: " + JSON.stringify(ret) + JSON.stringify(err)});
    });
}, 3000);
