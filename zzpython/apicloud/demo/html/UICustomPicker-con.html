UICustomPicker 模块模块是一个自定义内容选择器。

var UICustomPicker = api.require('UICustomPicker');
UICustomPicker.open({
    rect: {
        x: 50,
        y: 280,
        w: 180,
        h: 180
    },
    styles: {
        bg: 'rgba(0,0,80,0.8)',
        normalColor: 'rgb(0,0,120)',
        selectedColor: 'rgb(120,0,0)',
        selectedSize: 25,
        tagColor: 'rgb(0,120,0)',
        tagSize: 28
    },
     data: [{
        tag: '时',
        scope: '0-23'
       
    },
    {
        tag: '分',
        scope: '0-59'
      
    }
    ],
     rows: 5,
     autoHide:false
   
}, function(ret, err){
    alert(JSON.stringify(ret));
});
